```cpp
        if (arr[j] <= pivot) {
            i++;
            swap(arr[i], arr[j]);  // (2) pivot보다 작은 원소를 왼쪽 영역으로 이동
        }
    }
    swap(arr[i + 1], arr[high]);   // (3)
    return i + 1;                  // (4)
}

void quickSort(int arr[], int low, int high) {
    if (low < high) {              // (5)
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);      // (6)
        quickSort(arr, pi + 1, high);     // (7)
    }
}
```

**(8)** 최악의 경우: 이미 정렬된(오름차순 또는 내림차순) 배열에서 피벗이 항상 최솟값 또는 최댓값이 될 때 → **O(n²)**

</details>
